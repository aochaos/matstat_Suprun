def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        swapped = False
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
                swapped = True
        if not swapped:
            break

def input_array():
    n = int(input("   : "))
    arr = []
    for i in range(n):
        element = int(input(f"  {i+1}: "))
        arr.append(element)
    return arr
